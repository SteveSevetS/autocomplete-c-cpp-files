class Parameter
{
    type: string;
    name: string;
}

class Function
{
    type: string;
    functionName: string;
    parameters?: Parameter[];
}

class Class
{
    parentClass?: string[]; //string that contains the parent class already formatted
    type?: string;
    className: string;
    functions: Function[];
}

class Header
{
    namespaceName?: string;
    classes?: Class[];
    functions?: Function[];
    fileName: string;
    includes?: string[];
}
